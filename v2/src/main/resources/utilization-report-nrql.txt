SELECT max(cpuRequestedCores) as requestedCPU, max(cpuUsedCores/cpuRequestedCores*100) as maxCpuUtilizationPercentage, average(cpuUsedCores/cpuRequestedCores*100) as averageCpuUtilizationPercentage, percentile(cpuUsedCores/cpuRequestedCores*100, 50, 80, 99) as cpuUtilizationPercentiles, max(memoryRequestedBytes) as requestedMemory, max(memoryUsedBytes/memoryRequestedBytes*100) as maxMemoryUtilizationPercentage, average(memoryUsedBytes/memoryRequestedBytes*100) as averageMemoryUtilizationPercentage, percentile(memoryUsedBytes/memoryRequestedBytes*100, 50, 80, 99) as memoryUtilizationPercentiles FROM K8sContainerSample WHERE label.app IS NOT NULL FACET clusterName, label.app LIMIT 1000 SINCE 1 week ago
